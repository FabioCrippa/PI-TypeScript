Developer Dictionary - SQL

Create Database

A instrução CREATE DATABASE é usada para criar um novo banco de dados SQL.

CREATE DATABASE nomeBancodados;

Use

Aponta qual banco de dados será utilizado

USE nomeBancoDados;

Create table

Cria uma tabela

CREATE TABLE nomeTabela ( ID INT PRIMARY KEY AUTO_INCREMENT,
NOME VARCHAR(60) NOT NULL,
CPF CHAR(11) NOT NULL,
DATANASCIMENTO DATE NOT NULL,
ATIVO BIT NOT NULL,
)

PRIMARY KEY -- significa chave primária, ou seja, id único.
AUTO_INCREMENT -- significa incremento automático do id.
VARCHAR -- significa uma string com um número determinado de caracteres.
NOT NULL -- preenchimento obrigatório, ou seja, nao pode ser nulo.
FLOAT -- numero com virgula
FOREIGN KEY -- chave estrangeira que irá relacionar com uma outra tabela

Insert into

Insere dados na tabela

INSERT INTO nomeTabela (nomeColuna1, nomeColuna2, ativo, ...)
VALUES ("string", "number", booleano)

Drop database

Deleta um banco de dados

DROP DATABASE nomeBancoDados;

Drop table

Deleta uma tabela

DROP TABLE nomeTabela;

Update...Set...Where

Atualiza os dados da tabela

UPDATE nomeTabela SET nomeColuna = "dadosAtualizados";
WHERE ID = numeroId

Delete... From...Where

Deleta registro da tabela

DELETE FROM nomeTabela WHERE ID = numeroId;

Truncate table

Apaga todos os registros da tabela

TRUNCATE TABLE nomeTabela;

Relacionar tabelas - Foreign key

Relacionando a tabela1 com a tabela2

CREATE TABLE produtos ( ID INT PRIMARY KEY AUTO_INCREMENT,
DESCRICAO VARCHAR (100) NOT NULL,
QTDESTOQUE FLOAT NOT NULL,
PRECO FLOAT NOT NULL,
CATEGORIAID INT NOT NULL,
FOREIGN KEY (CATEGORIAID) REFERENCES CATEGORIAS (ID)
)

Insert into

Insere dados na tabela relacionada

INSERT INTO nomeTabela1(nomeColunaTabela2);
VALUES ("frios")

Select...From

Fazer pesquisa no banco de dados

SELECT nomeColuna FROM nomeTabela

Select * From

Busca todas as colunas de uma tabela

SELECT * FROM nomeTabela
